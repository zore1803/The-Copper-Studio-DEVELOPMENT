<script>
        function toggleSwitch(btn) {
            const circle = btn.querySelector('div');
            const isActive = btn.classList.contains('bg-primary');
            
            if (isActive) {
                btn.classList.remove('bg-primary');
                btn.classList.add('bg-secondary-container');
                circle.classList.remove('left-5');
                circle.classList.add('left-1');
            } else {
                btn.classList.remove('bg-secondary-container');
                btn.classList.add('bg-primary');
                circle.classList.remove('left-1');
                circle.classList.add('left-5');
            }
        }

        // Search bar micro-interaction
        const searchInput = document.querySelector('input[type="text"]');
        searchInput.addEventListener('focus', () => {
            searchInput.parentElement.classList.add('scale-[1.02]');
            searchInput.parentElement.classList.add('transition-transform');
        });
        searchInput.addEventListener('blur', () => {
            searchInput.parentElement.classList.remove('scale-[1.02]');
        });
    </script>
